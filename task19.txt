# Service
apiVersion: v1
kind: Service
metadata:
  name: repo1-svc
  namespace: i-ikromov
spec:
  selector:
    app: repo1-app
  ports:
    - protocol: TCP
      port: 80
      targetPort: 80

---
# Deploy
apiVersion: apps/v1
kind: Deployment
metadata:
  labels:
    app: repo1-app
  name: repo1-app
  namespace: i-ikromov
spec:
  replicas: 1
  selector:
    matchLabels:
      app: repo1-app
  strategy: {}
  template:
    metadata:
      labels:
        app: repo1-app
    spec:
      containers:
      - image: ikromovpro/repo-1:v1
        name: repo1-app
        resources:
          requests:
            cpu: "100m"
            memory: "64Mi"
          limits:
            cpu: "200m"
            memory: "128Mi"
        envFrom:
          - configMapRef:
              name: repo-info
          - secretRef:
              name: repo-secret
        env:
          - name: POD_IP
            valueFrom:
              fieldRef:
                fieldPath: status.podIP
         
          - name: NODE_NAME
            valueFrom:
              fieldRef:
                fieldPath: spec.nodeName
         
          - name: CPU
            valueFrom:
              resourceFieldRef:
                resource: requests.cpu

          - name: MEMORY
            valueFrom:
              resourceFieldRef:
                resource: limits.memory

status: {}
